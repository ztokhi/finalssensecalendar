<script>

(function(){

  const qs=new URLSearchParams(location.search);

  const iso=qs.get('date'); // optional: ?date=YYYY-MM-DD

  const now=iso?new Date(iso+'T00:00:00'):new Date();

  const wk=now.toLocaleDateString(undefined,{weekday:'long'}).toUpperCase();

  const mo=now.toLocaleDateString(undefined,{month:'long'}).toUpperCase();

  const d=now.getDate(); const y=now.getFullYear();

  document.getElementById('wk').textContent=wk;

  document.getElementById('d').textContent=d;

  document.getElementById('full').textContent=`${mo} ${d}, ${y}`;

})();

</script>
